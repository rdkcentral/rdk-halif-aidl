# @brief Build Android Binder SDK for RDK HAL AIDL modules
#
# This CMakeLists wraps the linux_binder_idl project and redirects
# output to out/target/ directory for use by HAL modules.
#
# Usage:
#   cmake -S build-tools/linux_binder_idl -B build/binder \
#         -DCMAKE_INSTALL_PREFIX=$PWD/out/target
#   cmake --build build/binder -j4
#   cmake --install build/binder
#
# This file is just documentation - use install_binder.sh script instead.
#
# Output:
#   out/target/lib/       - Binder runtime libraries
#   out/target/include/   - Binder headers
#   out/target/bin/       - aidl compiler and tools

message(WARNING 
    "This file is for documentation only.\n"
    "To build the binder SDK, use the install_binder.sh script:\n"
    "  ./install_binder.sh\n"
    "\n"
    "Or manually:\n"
    "  cmake -S build-tools/linux_binder_idl -B build/binder -DCMAKE_INSTALL_PREFIX=\$PWD/out/target\n"
    "  cmake --build build/binder -j4\n"
    "  cmake --install build/binder\n")

message(STATUS "")
message(STATUS "Binder SDK will be installed to: ${SDK_OUTPUT_DIR}")
message(STATUS "Run 'cmake --build <build-dir>' to build and stage the SDK")
message(STATUS "")
